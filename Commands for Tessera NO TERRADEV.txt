python3 << 'PYEOF'
import requests, json, time, os

results = []
packets = sorted(os.listdir("metadata_packets"))[:20]

for i, pf in enumerate(packets):
    with open(f"metadata_packets/{pf}") as f:
        meta = json.load(f)

    t0 = time.perf_counter()
    try:
        r = requests.post(
            "http://localhost:8080/generate",
            json={
                "user_id": meta["id"],
                "context": {"metadata": meta},
                "base_model": "meta-llama/Meta-Llama-3-8B-Instruct",
                "target_rank": 16,
                "response_format": "base64"
            },
            timeout=120
        )
        latency = (time.perf_counter() - t0) * 1000

        if r.status_code == 200:
            d = r.json()
            hit = d.get("cache_hit", False)
            results.append({
                "id": meta["id"],
                "latency_ms": latency,
                "cache_hit": hit,
                "generation_latency_ms": d.get("generation_latency_ms", 0),
            })
            print(f"  [{i+1}/20] {meta['id']}: {latency:.0f}ms ({'HIT' if hit else 'MISS'})")
        else:
            print(f"  [{i+1}/20] {meta['id']}: HTTP {r.status_code} — {r.text[:80]}")
            results.append({"id": meta["id"], "latency_ms": -1, "error": r.text[:100]})
    except Exception as e:
        print(f"  [{i+1}/20] {meta['id']}: {e}")
        results.append({"id": meta["id"], "latency_ms": -1, "error": str(e)})

with open("./results/tessera_latency.json", "w") as f:
    json.dump(results, f, indent=2)

valid = [r for r in results if r["latency_ms"] > 0]
hits = [r for r in valid if r.get("cache_hit")]
misses = [r for r in valid if not r.get("cache_hit")]
print(f"\nValid: {len(valid)}/20")
if misses: print(f"Cache miss avg: {sum(r['latency_ms'] for r in misses)/len(misses):.0f}ms")
if hits:   print(f"Cache hit avg:  {sum(r['latency_ms'] for r in hits)/len(hits):.0f}ms")
PYEOF

import json, glob, os
import matplotlib

import matplotlib.pyplot as plt

    baseline = json.load(f)

for task, score in baseline.items():
    print(f"  {task}: {score:.3f}")

# Compute deltas across all adapters
deltas = []
for adapter_dir in sorted(glob.glob("./results/adapters/*")):
    adapter_id = os.path.basename(adapter_dir)
    result_files = glob.glob(f"{adapter_dir}/*/results.json")
    if not result_files:
        continue
    with open(result_files[0]) as f:
        data = json.load(f)
    for task, scores in data.get("results", {}).items():
        adapter_score = scores.get("acc,none", scores.get("acc", 0))
        baseline_score = baseline.get(task, 0)
        delta = adapter_score - baseline_score
        deltas.append({
            "adapter_id": adapter_id,
            "domain": adapter_id.rsplit("_", 1)[0],
            "task": task,
            "baseline_score": round(baseline_score, 4),
            "adapter_score": round(adapter_score, 4),
            "delta": round(delta, 4),
            "delta_pct": round(delta * 100, 2),
        })

with open("./results/all_deltas.json", "w") as f:
    json.dump(deltas, f, indent=2)

print(f"\nTotal evaluations: {len(deltas)}")

if deltas:
    # ── Figure 1: Quality Delta by Domain ──
    domain_deltas = {}
    for d in deltas:
        domain_deltas.setdefault(d["domain"], []).append(d["delta_pct"])

    domains = sorted(domain_deltas.keys())
    avg_deltas = [np.mean(domain_deltas[d]) for d in domains]
    colors = ["#27ae60" if x >= 0 else "#c0392b" for x in avg_deltas]

    fig, ax = plt.subplots(figsize=(16, 7))
    bars = ax.bar(range(len(domains)), avg_deltas, color=colors,

    ax.axhline(y=0, color="black", linewidth=1.5)
    ax.set_xticks(range(len(domains)))
    ax.set_xticklabels(domains, rotation=45, ha="right", fontsize=9)
    ax.set_ylabel("Avg Accuracy Delta (%)", fontsize=12)
    ax.set_title("Tessera: Quality Delta by Domain\n(100 adapters vs Llama-3-8B baseline)", fontsize=13)
    ax.grid(axis="y", alpha=0.3)
    for bar, val in zip(bars, avg_deltas):
        ypos = bar.get_height() + 0.1 if val >= 0 else bar.get_height() - 0.4
        ax.text(bar.get_x() + bar.get_width()/2, ypos,
                f"{val:+.1f}%", ha="center", fontsize=7)
    plt.tight_layout()
    plt.savefig("figures/figure1_quality_delta_by_domain.png", dpi=150, bbox_inches="tight")

        plt.close()
        print("Figure 3 saved")

    # ── Figure 4: Summary table ──
    pos = len([d for d in deltas if d["delta"] > 0.005])
    neg = len([d for d in deltas if d["delta"] < -0.005])
    neu = len(deltas) - pos - neg
    avg = np.mean(all_pct)

    plt.close()
    print("Figure 4 saved")

    print(f"\n{'═'*50}")
    print("TESSERA BENCHMARK RESULTS")
